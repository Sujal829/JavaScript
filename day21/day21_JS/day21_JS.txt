Callback:
callback is a function passed as an argument to another function,
which is then executed after the first function completes it's task.



function getNameFromDB(cb){
    // DB connection and retrive name from DB
    let nm = 'ABC';
    cb(nm);
}

function display(name){
    console.log('recevied name from first function operation ', name);
}

getNameFromDB(display);
